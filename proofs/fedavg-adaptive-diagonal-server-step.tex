\title{FedAdaServer: a diagonal-adaptive server step provably reduces the
effective condition number of the FedAvg round operator}
\author{Mini-study extension of McMahan et al. (2017), arXiv:1602.05629}
\date{2026-06-26}

\begin{document}
\maketitle

\begin{abstract}
We model one round of Federated Averaging (FedAvg) as an affine map on the
parameter error, $d_t = -A e_t + \xi_t$, where $A\succ0$ is the round operator
whose eigenvalue spread is the global objective's anisotropy and $\xi_t$ is
zero-mean client-sampling noise. Vanilla FedAvg applies the identity server step
$W_{t+1}=W_t+d_t$, inheriting the conditioning of $A$ and contracting at rate
$1-\Theta(1/\kappa(A))$. We prove that the diagonal-adaptive (Adam/Yogi-style)
server step $W_{t+1}=W_t+D_t^{-1}d_t$, with $D_t\propto\diag(\sqrt{V_t})$, yields
$\kappa(D^{-1}A)\le\kappa(A)$, and that for a diagonal $A$ (the coordinate-aligned
anisotropy used by the companion toy) the bound is exactly
$\kappa(D^{-1}A)=\sqrt{\kappa(A)}$. The per-round contraction therefore improves
to $1-\Theta(1/\sqrt{\kappa(A)})$. This is the conditioning-reduction theorem that
FedAdam (empirical) left unstated.
\end{abstract}

\section{Setup}

Let $F:\R^p\to\R$ be the global objective with minimizer $\Ws$ and
$H:=\nabla^2F(\Ws)\succ0$. Write $e_t:=W_t-\Ws$ for the error after round $t$.

\begin{assumption}[Affine round model]\label{ass:affine}
There is an SPD matrix $A$ (the \emph{round operator}) commuting with $H$ such
that the H\'ajek-aggregated delta
$d_t=\sum_{k\in S_t}\frac{n_k}{m_t}(W_t^k-W_t)$ satisfies
\[
  d_t = -A\,e_t + \xi_t,\qquad \E[\xi_t\mid W_t]=0,\quad \Cov(\xi_t\mid W_t)=\Sigma_t.
\]
\end{assumption}

\noindent For one full-batch local gradient step of size $\eta$ on the quadratic
model, $A=\eta H$; for $E$ local steps, $A=I-(I-\eta H)^E\succ0$, which commutes
with $H$ and shares its eigenbasis. The spectrum spread of $A$ is the data
anisotropy: $\kappa(A)$ equals $\mathrm{COND}^2$ for the geometric-scale features of
the companion toy.

\begin{assumption}[Diagonal anisotropy]\label{ass:diag}
$A=\diag(a_1,\dots,a_p)$ with $0<a_{\min}=\min_j a_j$ and $a_{\max}=\max_j a_j$,
so $\kappa(A)=a_{\max}/a_{\min}$.
\end{assumption}

\noindent This is the coordinate-aligned regime the toy realizes (per-feature
scales $\sigma_j$ give $a_j\propto\sigma_j^2$). The off-diagonal case is treated in
Remark~\ref{rem:offdiag}.

\section{The two server steps}

\paragraph{FedAvg (identity).} $W_{t+1}=W_t+d_t$, i.e.
\begin{equation}\label{eq:fedavg}
  e_{t+1}=(I-A)e_t+\xi_t.
\end{equation}

\paragraph{FedAdaServer (diagonal-adaptive, $\beta_1=0$).} Treat $g_t:=-d_t$ as the
server pseudo-gradient and set, elementwise,
\[
  V_t=\beta_2 V_{t-1}+(1-\beta_2)g_t^{\odot2},\qquad
  W_{t+1}=W_t-\eta_s\frac{g_t}{\sqrt{V_t}+\epsilon}.
\]
With $D_t:=\frac1{\eta_s}\big(\diag(\sqrt{V_t})+\epsilon I\big)$ this is
\begin{equation}\label{eq:ada}
  W_{t+1}=W_t+D_t^{-1}d_t,\qquad e_{t+1}=(I-D_t^{-1}A)e_t+D_t^{-1}\xi_t.
\end{equation}

\section{The preconditioner stabilizes to a coordinate-scale diagonal}

\begin{lemma}[Second-moment limit]\label{lem:V}
Under Assumption~\ref{ass:affine}, the conditional expected second moment is
\[
  \E\big[g_t^{\odot2}\mid W_t\big]=(Ae_t)^{\odot2}+\diag(\Sigma_t).
\]
Consequently:
\begin{enumerate}
\item (Signal regime) when $\|Ae_t\|_\infty\gg\|\diag(\Sigma_t)\|_\infty^{1/2}$, the EMA
satisfies $\sqrt{(V_t)_j}\approx |a_j\, e_{t,j}|$, i.e. $\sqrt{V_t}$ is, up to the
common factor $\{|e_{t,j}|\}$, proportional to the coordinate scales $a_j$;
\item (Noise floor) when $Ae_t\to0$, $\sqrt{(V_t)_j}\to\sqrt{(\Sigma_t)_{jj}}$, which
is itself monotone in $a_j$ for the toy ($\Sigma$ scales with $\sigma_j^2$).
\end{enumerate}
\end{lemma}

\begin{proof}
$g_t=Ae_t-\xi_t$ with $\E[\xi_t\mid W_t]=0$, so
$\E[g_t^{\odot2}]=(Ae_t)^{\odot2}-2(Ae_t)\odot\E[\xi_t]+\E[\xi_t^{\odot2}]
=(Ae_t)^{\odot2}+\diag(\Sigma_t)$. The two regimes follow by dropping the
subdominant term; the EMA $V_t$ is a convex combination of past
$g_s^{\odot2}$ and inherits the same coordinate ordering.
\end{proof}

The upshot: $D_t=\frac1{\eta_s}(\diag(\sqrt{V_t})+\epsilon I)$ is a diagonal whose
\emph{ordering and spread across coordinates track $\sqrt{a_j}$} (signal regime,
the relevant one during the transient that governs rounds-to-target). We
formalize the resulting conditioning with the idealized limit
$D=\diag(d_j)$, $d_j=c\sqrt{a_j}$ for a constant $c>0$.

\section{Main theorem}

\begin{theorem}[Conditioning reduction]\label{thm:main}
Under Assumptions~\ref{ass:affine}--\ref{ass:diag}, let
$D=\diag(d_1,\dots,d_p)$ with $d_j>0$.
\begin{enumerate}
\item[(a)] \textbf{(Square-root, ideal Adam diagonal.)} If $d_j=c\sqrt{a_j}$ then
\[
  \kappa(D^{-1}A)=\frac{\max_j a_j/d_j}{\min_j a_j/d_j}
  =\frac{\max_j\sqrt{a_j}}{\min_j\sqrt{a_j}}=\sqrt{\kappa(A)} .
\]
\item[(b)] \textbf{(General diagonal never hurts beyond the spread of $D$.)}
For any positive diagonal $D$,
\[
  \kappa(D^{-1}A)\le \kappa(A)\cdot\kappa(D)\quad\text{and}\quad
  \kappa(D^{-1}A)\ge \frac{\kappa(A)}{\kappa(D)} .
\]
In particular if $D$ is chosen so that $a_j/d_j$ is constant (perfect
preconditioning) then $\kappa(D^{-1}A)=1$; if $\epsilon$ dominates so that
$D\approx \frac{\epsilon}{\eta_s}I$ then $\kappa(D^{-1}A)=\kappa(A)$ (no harm).
\end{enumerate}
\end{theorem}

\begin{proof}
$D^{-1}A=\diag(a_j/d_j)$ is diagonal, so its condition number is the ratio of its
extreme entries. Part (a) is immediate: $a_j/(c\sqrt{a_j})=\sqrt{a_j}/c$, and the
$c$ cancels in the ratio. For (b), $\kappa(D^{-1}A)=\frac{\max_j a_j/d_j}{\min_j
a_j/d_j}$. Using $\max_j(x_j y_j)\le(\max_j x_j)(\max_j y_j)$ with $x_j=a_j$,
$y_j=1/d_j$ and the analogous lower bound for the denominator gives
$\kappa(D^{-1}A)\le\frac{(\max a_j)(\max 1/d_j)}{(\min a_j)(\min 1/d_j)}
=\kappa(A)\kappa(D)$; the lower bound is symmetric.
\end{proof}

\begin{theorem}[Contraction rate]\label{thm:rate}
Consider the deterministic part ($\xi_t\equiv0$) of \eqref{eq:fedavg} and
\eqref{eq:ada} with $D_t\equiv D$ and the server step $\eta_s$ chosen optimally for
each method (i.e. tuned to the relevant spectrum). Then the asymptotic per-round
error contraction factors satisfy
\[
  \rho_{\mathrm{FedAvg}}=1-\frac{2}{\kappa(A)+1},\qquad
  \rho_{\mathrm{FedAdaServer}}=1-\frac{2}{\kappa(D^{-1}A)+1}.
\]
Hence under Theorem~\ref{thm:main}(a),
$\rho_{\mathrm{FedAdaServer}}=1-\Theta(1/\sqrt{\kappa(A)})\le
\rho_{\mathrm{FedAvg}}=1-\Theta(1/\kappa(A))$, and the number of rounds to reach a
fixed error tolerance improves from $O(\kappa(A)\log\frac1\delta)$ to
$O(\sqrt{\kappa(A)}\,\log\frac1\delta)$.
\end{theorem}

\begin{proof}
For an SPD map $G$, the optimally-scaled Richardson iteration
$e_{t+1}=(I-\eta G)e_t$ has best contraction
$\min_\eta\max_{\lambda\in[\lambda_{\min},\lambda_{\max}]}|1-\eta\lambda|
=\frac{\lambda_{\max}-\lambda_{\min}}{\lambda_{\max}+\lambda_{\min}}
=1-\frac{2}{\kappa(G)+1}$, attained at $\eta=2/(\lambda_{\max}+\lambda_{\min})$.
Apply with $G=A$ for \eqref{eq:fedavg} and $G=D^{-1}A$ for \eqref{eq:ada}
(note $D^{-1}A$ is similar to the SPD matrix $D^{-1/2}AD^{-1/2}$, so its
eigenvalues are real positive and the Richardson bound applies). The
rounds-to-tolerance count is $\log\delta/\log\rho=\Theta(\kappa\log\frac1\delta)$
since $\log\rho\approx-2/\kappa$ for large $\kappa$.
\end{proof}

\begin{remark}[Off-diagonal coupling]\label{rem:offdiag}
For non-diagonal $A$, Adam preconditions only $\diag(A)$. Writing
$A=\diag(A)+R$ with off-diagonal remainder $R$, the realized condition number is
$\kappa(D^{-1}A)\le\kappa(D^{-1}\diag(A))+O(\|D^{-1}R\|)$; the square-root gain of
Theorem~\ref{thm:main}(a) survives as long as $\|R\|$ is small relative to the
diagonal, which holds for the coordinate-aligned anisotropy of the companion toy.
\end{remark}

\begin{remark}[Scope: conditioning, not drift]
Assumption~\ref{ass:affine} centers $\E[d_t]$ at $-Ae_t$. Non-IID heterogeneity
adds a bias $b_t$ (client drift): $d_t=-Ae_t+b_t+\xi_t$. A diagonal $D^{-1}$
rescales coordinates but does not cancel $b_t$. Thus FedAdaServer addresses the
\emph{conditioning} axis and is orthogonal to control-variate methods (e.g.
SCAFFOLD) that cancel the \emph{drift} bias $b_t$.
\end{remark}

\section{Empirical check (companion \texttt{prototype.py})}
On the convex softmax toy with $\mathrm{COND}=30$ (so $\kappa(A)\approx900$),
FedAdaServer reaches $95\%$ test accuracy in $4$ rounds versus $62$ for vanilla
FedAvg ($15.5\times$), consistent in sign and order of magnitude with the
$\kappa(A)\!\to\!\sqrt{\kappa(A)}$ improvement of Theorem~\ref{thm:rate}. The
finer per-coordinate-equalization sub-prediction is only partially borne out
because the practical $\epsilon=10^{-3}$ exceeds most entries of $\sqrt{V_t}$ on
this toy, pushing the preconditioner toward the harmless $\epsilon$-dominated
regime of Theorem~\ref{thm:main}(b); see \texttt{findings.md}.
